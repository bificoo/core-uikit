# ControlActiveKey

<Preview>
  <Story name="ControlActiveKey">
    {() => {
      const [active, setActive] = useState("Tab1");
      const style = {
        alignItems: "center",
        backgroundColor: "#f4f5f7",
        borderRadius: "3px",
        color: "#6b778c",
        display: "flex",
        fontSize: "4em",
        fontWeight: 500,
        flexGrow: 1,
        justifyContent: "center",
        padding: "32px",
      };
      return (
        <div>
          <div style={{ marginBottom: "20px" }}>
            <Button
              disabled={active === "Tab1"}
              onClick={() => setActive("Tab1")}
              style={{ marginRight: "10px" }}
            >
              set Tab1 active
            </Button>
            <Button
              disabled={active === "Tab2"}
              onClick={() => setActive("Tab2")}
              style={{ marginRight: "10px" }}
            >
              set Tab2 active
            </Button>
            <Button
              disabled={active === "Tab3"}
              onClick={() => setActive("Tab3")}
            >
              set Tab3 active
            </Button>
          </div>
          <Tabs activeKey={active} defaultActiveKey={active}>
            <Tabs.Tab title="數據指標" eventKey="Tab1" />
            <Tabs.Tab title="圖像報表" eventKey="Tab2" />
            <Tabs.Tab title="用戶清單" eventKey="Tab3" />
          </Tabs>
          <div style={style}>Content</div>
        </div>
      );
    }}
  </Story>
</Preview>
